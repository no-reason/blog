		scanf("%d",&a[i]);
	}

		if (sum>=s){

			l++;
		}
		if (sum<s){
			sum+=a[r];
			r++; 
		}

	else cout<<ans<<endl;

}

void qsort (int l,int r){
	int mid=a[(l+r)/2],i=l,j=r;
	while (i<j) {
		while (a[i]<mid) i++;
		while(a[j]>mid) j--;
		if(i<=j){ swap(a[i],a[j]);i++,j--;}

int n;

int b[N];

void msort(int l,int r){
	if (l==r)
		return ; 
	int mid=(l+r)>>1;
	msort(l,mid);
	msort(mid+1,r);
	int pl=l;
	int pr=mid+1;
	int pb=l;
	while (pl<=mid||pr<=r){
		if (pl>mid){
			b[pb++]=a[pr++];
		}
		else if (pr>r) b[pb++]=a[pl++];
		else{
			if (a[pl]<=a[pr]) b[pb++]=a[pl++];
			else{
				b[pb++]=a[pr++];
			} 
		}
	}
	for (int i=l;i<=r;i++)
		a[i]=b[i];
}

#include<iostream>

using namespace std;

int n;
const int N=5e5+10;
int a[N];
int b[N];
long long ans=0;
void msort(int l,int r){
	if (l==r)
		return ; 
	int mid=(l+r)>>1;
	msort(l,mid);
	msort(mid+1,r);
	int pl=l;
	int pr=mid+1;
	int pb=l;
	while (pl<=mid||pr<=r){
		if (pl>mid){
			b[pb++]=a[pr++];
		}
		else if (pr>r) b[pb++]=a[pl++];
		else{
			if (a[pl]<=a[pr]) b[pb++]=a[pl++];
			else{
				b[pb++]=a[pr++]; ans+=mid-pl+1; 
			} 
		}
	}
	for (int i=l;i<=r;i++)
		a[i]=b[i];
}
int main(){
	cin>>n;
	for (int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	msort(1,n);
